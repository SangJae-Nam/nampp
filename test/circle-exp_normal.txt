class circle extends object
 field int r
 field int i
 field int j
 field string sline
 field string color

method void initialize(rr: int, ii: int, jj: int, outline: string, cc: string)
 begin
 set r = rr;
 set i = ii
 set j = jj
 set sline = outline
 set color = cc
 end
method void setCircle(rr: int, ii: int, jj: int, outline: string, cc: string)
 begin
 set r = rr;
 set i = ii
 set j = jj
 set sline = outline
 end
 method void setColor(cc: string)
  begin
   set color = cc;
  end
 method void setSline(outline: string)
  begin
   set sline = outline;
  end
 method void drawCircle()
  begin
   circle(r,i,j,sline,color)
  end
 let c1 = new circle(5,15,30,"solid", "red")

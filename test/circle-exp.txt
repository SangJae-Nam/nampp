% circle-exp  test - Normal case
class circle extends object
 field int r
 field int i
 field int j
 field bool sline
 field int red
 field int green
 field int blue

method void initialize(rr: int, ii: int, jj: int, outline: bool, rred: int, ggreen: int, bblue: int)
 begin
 set r = rr;
 set i = ii;
 set j = jj;
 set sline = outline;
 set red = rred;
 set green = ggreen;
 set blue = bblue
 end
 
method void setCircle(rr: int, ii: int, jj: int, outline: string, cc: string)
 begin
 set r = rr;
 set i = ii;
 set j = jj;
 set sline = outline;
 set red = rred;
 set green = ggreen;
 set blue = bblue
 end

method void drawCircle()
  begin
   if sline then
     circle(r, i, j, 0, red, green, blue)
   else
     circle(r, i, j, 1, red, green, blue)
  end
  
let c1 = new circle(5, 15, 30, true, 255, 0, 255) in
let c2 = new circle(20, 30, 50, false, 0, 255, 255) in
 send c1 drawCircle()

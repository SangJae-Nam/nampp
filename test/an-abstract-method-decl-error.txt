% an-abstract-method-decl 
% =========================================================
% an-abstract-method-decl  test - Normal case
interface inter
 method void sum(x: int)

class cl extends object implements inter
 field int i

 method void initialize(x: int)
   set i = x

 method void sum(x: int)
   set i = +(i,x)

%%오류 - 리턴타입의 정의와 실제 타입이 같지 않음
 method void get_i()
   i

let o1 = new cl(1000) in
  begin
    send o1 sum(999);
    send o1 get_i()
  end
